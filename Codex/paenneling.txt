def dbg(msg):
    print("[PanelTool] {}".format(msg))


def force_refresh():
    try:
        cmds.refresh(force=True)
    except Exception:
        pass


def _valid_components(items):
    out = []
    for item in items or []:
        base = item.split(".")[0]
        if cmds.objExists(base):
            out.append(item)
    return out


def show_step(title, message, selection=None):
    dbg("=" * 80)
    dbg(title)
    dbg(message)

    valid = _valid_components(selection or [])
    try:
        if valid:
            cmds.select(valid, r=True)
        else:
            cmds.select(clear=True)
    except Exception:
        pass

    force_refresh()

    result = cmds.confirmDialog(
        title=title,
        message=message,
        button=["Continue", "Stop"],
        defaultButton="Continue",
        cancelButton="Stop",
        dismissString="Stop"
    )

    if result != "Continue":
        raise RuntimeError("Workflow stoppé par l'utilisateur à l'étape : {}".format(title))


def show_info(title, message, selection=None):
    dbg("=" * 80)
    dbg(title)
    dbg(message)

    valid = _valid_components(selection or [])
    try:
        if valid:
            cmds.select(valid, r=True)
        else:
            cmds.select(clear=True)
    except Exception:
        pass

    force_refresh()

    cmds.confirmDialog(
        title=title,
        message=message,
        button=["OK"],
        defaultButton="OK"
    )

def _get_internal_uv_seam_edges(obj):
    """
    Seam UV = même edge 3D mais UV IDs différents sur les 2 faces voisines.
    """
    selection_list = om.MSelectionList()
    selection_list.add(obj)
    dag_path = selection_list.getDagPath(0)

    mesh = om.MFnMesh(dag_path)
    seam_edges = []

    edge_it = om.MItMeshEdge(dag_path)

    while not edge_it.isDone():
        edge_index = edge_it.index()
        faces = edge_it.getConnectedFaces()

        if len(faces) != 2:
            edge_it.next()
            continue

        v1 = edge_it.vertexId(0)
        v2 = edge_it.vertexId(1)

        uv_ids_per_face = []

        for face_id in faces:
            face_vertices = mesh.getPolygonVertices(face_id)

            i1 = -1
            i2 = -1

            for i in range(len(face_vertices)):
                if face_vertices[i] == v1:
                    i1 = i
                if face_vertices[i] == v2:
                    i2 = i

            if i1 == -1 or i2 == -1:
                continue

            try:
                uv_id1 = mesh.getPolygonUVid(face_id, i1)
                uv_id2 = mesh.getPolygonUVid(face_id, i2)
            except Exception:
                continue

            uv_ids_per_face.append((uv_id1, uv_id2))

        if len(uv_ids_per_face) == 2:
            if uv_ids_per_face[0] != uv_ids_per_face[1]:
                seam_edges.append("{}.e[{}]".format(obj, edge_index))

        edge_it.next()

    return seam_edges

def _face_center(face):
    verts = cmds.ls(cmds.polyListComponentConversion(face, ff=True, tv=True), fl=True) or []
    if not verts:
        return [0.0, 0.0, 0.0]

    pts = [cmds.xform(v, q=True, ws=True, t=True) for v in verts]
    cx = sum(p[0] for p in pts) / len(pts)
    cy = sum(p[1] for p in pts) / len(pts)
    cz = sum(p[2] for p in pts) / len(pts)
    return [cx, cy, cz]


def _distance_sq(a, b):
    return ((a[0] - b[0]) ** 2) + ((a[1] - b[1]) ** 2) + ((a[2] - b[2]) ** 2)


def _get_front_back_side_faces(obj, faces_before):
    """
    Après extrude sur toutes les faces :
    - back_faces = faces d'origine
    - front_faces = nouvelles faces les plus proches des centres d'origine
    - side_faces = reste des nouvelles faces
    """
    all_faces_now = cmds.ls(obj + ".f[*]", fl=True) or []
    faces_before = _valid_components(faces_before)
    before_set = set(faces_before)

    new_faces = [f for f in all_faces_now if f not in before_set]

    dbg("Faces avant extrude : {}".format(len(faces_before)))
    dbg("Faces après extrude : {}".format(len(all_faces_now)))
    dbg("Nouvelles faces : {}".format(len(new_faces)))

    if not new_faces:
        return faces_before, [], []

    before_centers = {f: _face_center(f) for f in faces_before}
    new_centers = {f: _face_center(f) for f in new_faces}

    remaining_new = set(new_faces)
    front_faces = []

    for old_face in faces_before:
        if not remaining_new:
            break

        oc = before_centers[old_face]
        best = min(
            remaining_new,
            key=lambda nf: _distance_sq(oc, new_centers[nf])
        )
        front_faces.append(best)
        remaining_new.remove(best)

    side_faces = list(remaining_new)
    back_faces = list(faces_before)

    return back_faces, front_faces, side_faces


def _get_connected_shells(faces):
    visited = set()
    shells = []
    face_set = set(faces)

    for f in faces:
        if f in visited:
            continue

        stack = [f]
        shell = []

        while stack:
            cur = stack.pop()
            if cur in visited:
                continue

            visited.add(cur)
            shell.append(cur)

            neighbors = cmds.ls(
                cmds.polyListComponentConversion(cur, ff=True, tf=True),
                fl=True
            ) or []

            for n in neighbors:
                if n in face_set and n not in visited:
                    stack.append(n)

        shells.append(shell)

    return shells


def _center(faces):
    verts = cmds.ls(cmds.polyListComponentConversion(faces, tv=True), fl=True) or []
    if not verts:
        return [0.0, 0.0, 0.0]

    pos = [cmds.xform(v, q=True, ws=True, t=True) for v in verts]
    return [sum(x) / len(x) for x in zip(*pos)]


def _pair_shells_by_center(front, back):
    pairs = []
    remaining_back = list(back)

    for f in front:
        if not remaining_back:
            break

        fc = _center(f)
        best = min(
            remaining_back,
            key=lambda b: sum((fc[i] - _center(b)[i]) ** 2 for i in range(3))
        )
        pairs.append((f, best))
        remaining_back.remove(best)

    return pairs


def _get_edge_vertices(edge):
    verts = cmds.ls(
        cmds.polyListComponentConversion(edge, fromEdge=True, toVertex=True),
        fl=True
    ) or []
    return [v for v in verts if ".vtx[" in v and cmds.objExists(v)]


def _get_edge_center(edge):
    verts = _get_edge_vertices(edge)
    if not verts:
        return [0.0, 0.0, 0.0]

    pts = [cmds.xform(v, q=True, ws=True, t=True) for v in verts]
    cx = sum(p[0] for p in pts) / len(pts)
    cy = sum(p[1] for p in pts) / len(pts)
    cz = sum(p[2] for p in pts) / len(pts)
    return [cx, cy, cz]


def _get_loop_center(edges):
    verts = cmds.ls(
        cmds.polyListComponentConversion(edges, fromEdge=True, toVertex=True),
        fl=True
    ) or []
    verts = [v for v in verts if ".vtx[" in v and cmds.objExists(v)]

    if not verts:
        return [0.0, 0.0, 0.0]

    pts = [cmds.xform(v, q=True, ws=True, t=True) for v in verts]
    cx = sum(p[0] for p in pts) / len(pts)
    cy = sum(p[1] for p in pts) / len(pts)
    cz = sum(p[2] for p in pts) / len(pts)
    return [cx, cy, cz]


def _is_open_border_edge(edge):
    faces = cmds.ls(
        cmds.polyListComponentConversion(edge, fromEdge=True, toFace=True),
        fl=True
    ) or []
    faces = [f for f in faces if ".f[" in f and cmds.objExists(f)]
    return len(faces) == 1


def _get_connected_edge_loops(edges):
    """
    Groupe des edges connectées par vertices.
    """
    edges = [e for e in (edges or []) if ".e[" in e and cmds.objExists(e)]
    if not edges:
        return []

    edge_to_verts = {}
    vert_to_edges = {}

    for edge in edges:
        verts = _get_edge_vertices(edge)
        edge_to_verts[edge] = verts

        for v in verts:
            vert_to_edges.setdefault(v, set()).add(edge)

    visited = set()
    loops = []

    for edge in edges:
        if edge in visited:
            continue

        stack = [edge]
        loop = []

        while stack:
            cur = stack.pop()
            if cur in visited:
                continue

            visited.add(cur)
            loop.append(cur)

            for v in edge_to_verts.get(cur, []):
                for neighbor in vert_to_edges.get(v, []):
                    if neighbor not in visited:
                        stack.append(neighbor)

        loops.append(loop)

    return loops


def _get_border_loops_from_face_set(face_set_name):
    """
    Convertit un set de faces en edges de périmètre, puis garde seulement
    les edges réellement ouvertes, et les groupe en loops.
    """
    if not cmds.objExists(face_set_name):
        return []

    faces = cmds.ls(cmds.sets(face_set_name, q=True), fl=True) or []
    faces = [f for f in faces if ".f[" in f and cmds.objExists(f)]

    if not faces:
        return []

    cmds.select(faces, r=True)
    mel.eval("ConvertSelectionToEdgePerimeter;")

    edges = cmds.ls(sl=True, fl=True) or []
    edges = [e for e in edges if ".e[" in e and cmds.objExists(e)]

    border_edges = [e for e in edges if _is_open_border_edge(e)]

    dbg("Face set {} -> perimeter edges : {} | open border edges : {}".format(
        face_set_name, len(edges), len(border_edges))
    )

    loops = _get_connected_edge_loops(border_edges)
    loops = [loop for loop in loops if len(loop) >= 3]

    dbg("Face set {} -> loops détectées : {}".format(face_set_name, len(loops)))

    return loops

def panel_start_workflow(state):
    try:
        sel = cmds.ls(sl=True, fl=True, long=True)
        edges = cmds.filterExpand(sel, sm=32)

        if not edges:
            cmds.warning("Sélectionne des edges.")
            return

        obj = edges[0].split(".e[")[0]

        if not cmds.objExists(obj):
            cmds.warning("Objet introuvable.")
            return

        short_name = obj.split("|")[-1]

        default_value = cmds.floatSliderGrp(
            state["ui"]["extrude_slider"], q=True, value=True
        )

        auto_bevel = cmds.checkBox(
            state["ui"]["auto_bevel_check"], q=True, value=True
        )

        state["started"] = False
        state["obj"] = obj
        state["short_name"] = short_name
        state["extrude_node"] = None
        state["is_negative"] = default_value < 0.0
        state["pending_is_negative"] = default_value < 0.0
        state["internal_restore"] = False
        state["safe_selection"] = [obj]
        state["bridge_nodes"] = []
        state["bevel_node"] = None

        dbg("Objet détecté : {}".format(obj))
        dbg("Edges sélectionnées : {}".format(len(edges)))

        show_step(
            "Step 1 - Sélection",
            "Objet : {}\nEdges sélectionnées : {}".format(short_name, len(edges)),
            edges[:300]
        )

        backup = cmds.duplicate(obj, name=short_name + "_backup#")[0]
        cmds.hide(backup)
        state["backup"] = backup

        dbg("Backup créé : {}".format(backup))
        show_step(
            "Step 2 - Backup",
            "Backup créé : {}".format(backup),
            [backup]
        )

        edge_set = short_name + "_panelEdges_set"
        if cmds.objExists(edge_set):
            cmds.delete(edge_set)
        cmds.sets(edges, name=edge_set)

        show_step(
            "Step 3 - Edge Set",
            "Set créé : {}\nEdges stockées : {}".format(edge_set, len(edges)),
            edges[:300]
        )

        cmds.select(obj, r=True)

        try:
            mel.eval("UVCameraBasedProjection;")
            dbg("UVCameraBasedProjection OK")
        except Exception:
            cmds.warning("UVCameraBasedProjection a échoué.")

        show_step("Step 4 - Camera Projection", "Projection UV faite.", [obj])

        try:
            mel.eval("UnfoldUV;")
            dbg("UnfoldUV OK")
        except Exception:
            dbg("UnfoldUV ignoré / échoué")

        show_step("Step 5 - Unfold", "Unfold terminé.", [obj])

        try:
            mel.eval("UVOrientShells;")
            dbg("UVOrientShells OK")
        except Exception:
            dbg("UVOrientShells ignoré / échoué")

        show_step("Step 6 - Orient Shells", "Orientation des shells terminée.", [obj])

        try:
            mel.eval("LayoutUV;")
            dbg("LayoutUV OK")
        except Exception:
            dbg("LayoutUV ignoré / échoué")

        show_step("Step 7 - Layout UV", "Layout UV terminé.", [obj])

        stored_edges = cmds.ls(cmds.sets(edge_set, q=True), fl=True) or []
        if stored_edges:
            cmds.select(stored_edges, r=True)
            try:
                cmds.polyMapCut(stored_edges)
                dbg("polyMapCut OK")
            except Exception:
                try:
                    mel.eval("performPolyMapCut;")
                    dbg("performPolyMapCut OK")
                except Exception:
                    cmds.warning("Cut UV a échoué.")
                    return

        show_step(
            "Step 8 - Cut UV",
            "Cut UV terminé.\nEdges coupées : {}".format(len(stored_edges)),
            stored_edges[:300]
        )

        faces_before = cmds.ls(obj + ".f[*]", fl=True) or []
        dbg("Faces avant extrude : {}".format(len(faces_before)))

        cmds.select(obj + ".f[*]", r=True)
        extrude_node = cmds.polyExtrudeFacet(
            localTranslateZ=default_value,
            keepFacesTogether=True
        )[0]

        state["extrude_node"] = extrude_node
        dbg("Extrude node : {}".format(extrude_node))

        show_step(
            "Step 9 - Extrude",
            "Extrude créé.\nDistance : {}".format(default_value),
            [obj]
        )

        if state["is_negative"]:
            cmds.select(obj, r=True)
            mel.eval("ReversePolygonNormals;")
            dbg("Normals inversées car extrude négative.")
            show_step(
                "Step 10 - Reverse Normals",
                "Normals inversées car distance négative.",
                [obj]
            )

        back_faces, front_faces, side_faces = _get_front_back_side_faces(obj, faces_before)

        dbg("Back faces : {}".format(len(back_faces)))
        dbg("Front faces : {}".format(len(front_faces)))
        dbg("Side faces : {}".format(len(side_faces)))

        front_set = short_name + "_panelFrontFaces_set"
        back_set = short_name + "_panelBackFaces_set"
        side_set = short_name + "_panelSideFaces_set"

        for s in [front_set, back_set, side_set]:
            if cmds.objExists(s):
                cmds.delete(s)

        if front_faces:
            cmds.sets(front_faces, name=front_set)
        if back_faces:
            cmds.sets(back_faces, name=back_set)
        if side_faces:
            cmds.sets(side_faces, name=side_set)

        show_step(
            "Step 11 - Front Faces",
            "Front faces détectées : {}".format(len(front_faces)),
            front_faces[:300] if front_faces else [obj]
        )

        show_step(
            "Step 12 - Back Faces",
            "Back faces détectées : {}".format(len(back_faces)),
            back_faces[:300] if back_faces else [obj]
        )

        show_step(
            "Step 13 - Side Faces",
            "Side faces détectées : {}".format(len(side_faces)),
            side_faces[:300] if side_faces else [obj]
        )

        seam_edges = _get_internal_uv_seam_edges(obj)

        seam_set = short_name + "_panelSeams_set"
        if cmds.objExists(seam_set):
            cmds.delete(seam_set)
        if seam_edges:
            cmds.sets(seam_edges, name=seam_set)

        dbg("Seams UV détectées : {}".format(len(seam_edges)))
        show_step(
            "Step 14 - UV Seams",
            "Seams UV détectées : {}".format(len(seam_edges)),
            seam_edges[:300] if seam_edges else [obj]
        )

        cmds.select(obj, r=True)
        state["safe_selection"] = [obj]

        panel_delete_side_faces(short_name)
        show_step(
            "Step 15 - Delete Side Faces",
            "Side faces supprimées.",
            [obj]
        )

        panel_detach(short_name)
        show_step(
            "Step 16 - Detach",
            "Panels détachés sur les seams.",
            [obj]
        )

        panel_bridge(short_name, state)
        show_step(
            "Step 17 - Bridge",
            "Bridge terminé.\nBridge nodes : {}".format(len(state["bridge_nodes"])),
            [obj]
        )

        state["started"] = True
        panel_enable_controls(state, True)

        if auto_bevel:
            panel_bevel(short_name, state)
            panel_force_initial_bevel_refresh(short_name, state)

            show_step(
                "Step 18 - Auto Bevel",
                "Bevel automatique appliqué.",
                [obj]
            )

        cmds.select(obj, r=True)

        show_info(
            "Workflow prêt",
            "Start terminé.\nTu peux maintenant ajuster l'extrude, bevel, validate ou revert.",
            [obj]
        )

    except Exception as e:
        cmds.warning(str(e))
        show_info(
            "Erreur",
            "Le workflow s'est arrêté.\n\n{}".format(e),
            [state.get("obj")] if state.get("obj") else None
        )

# EXTRUDE UPDATE
# =========================================================

def preview_extrude(state, value):
    extrude_node = state.get("extrude_node")
    if not extrude_node or not cmds.objExists(extrude_node):
        return

    cmds.setAttr(extrude_node + ".localTranslateZ", value)
    state["pending_is_negative"] = value < 0.0

    if not state["internal_restore"]:
        current_sel = cmds.ls(sl=True, long=True) or []
        if current_sel and len(current_sel) < 200:
            state["safe_selection"] = current_sel


def finalize_extrude(state, value):
    extrude_node = state.get("extrude_node")
    obj = state.get("obj")
    short_name = state.get("short_name")

    if not extrude_node or not cmds.objExists(extrude_node):
        return

    new_is_negative = value < 0.0
    old_is_negative = state["is_negative"]

    if new_is_negative != old_is_negative:
        try:
            state["internal_restore"] = True
            cmds.select(clear=True)
            cmds.select(obj, r=True)
            mel.eval("ReversePolygonNormals;")
            dbg("Normals inversées après changement de signe d'extrude.")
        except Exception as e:
            cmds.warning("Reverse failed: {}".format(e))
        finally:
            state["internal_restore"] = False

        state["is_negative"] = new_is_negative
        state["pending_is_negative"] = new_is_negative

        for bridge_node in state["bridge_nodes"]:
            if cmds.objExists(bridge_node):
                try:
                    cmds.setAttr(bridge_node + ".sourceDirection", 0)
                except Exception:
                    pass

        panel_rebuild_bevel(short_name, state)
    else:
        state["is_negative"] = new_is_negative
        state["pending_is_negative"] = new_is_negative

    try:
        state["internal_restore"] = True
        cmds.select(clear=True)

        safe_sel = state.get("safe_selection") or [obj]
        valid_sel = [x for x in safe_sel if cmds.objExists(x.split(".")[0])]
        if valid_sel:
            cmds.select(valid_sel, r=True)
        else:
            cmds.select(obj, r=True)
    except Exception:
        cmds.select(obj, r=True)
    finally:
        state["internal_restore"] = False

def panel_bridge(short_name, state):
    front_set = short_name + "_panelFrontFaces_set"
    back_set = short_name + "_panelBackFaces_set"

    front_loops = _get_border_loops_from_face_set(front_set)
    back_loops = _get_border_loops_from_face_set(back_set)

    if not front_loops or not back_loops:
        cmds.warning("Impossible de trouver les border loops front/back.")
        return

    dbg("Front loops : {}".format(len(front_loops)))
    dbg("Back loops : {}".format(len(back_loops)))

    remaining_back = list(back_loops)
    created_bridge_nodes = []
    used_edges = set()

    for front_loop in front_loops:
        if not remaining_back:
            break

        fc = _get_loop_center(front_loop)
        candidates = [b for b in remaining_back if len(b) == len(front_loop)]

        if not candidates:
            dbg("Aucune loop back compatible (edge count mismatch).")
            continue

        best_back = min(
            candidates,
            key=lambda b: _distance_sq(fc, _get_loop_center(b))
        )

        front_edges = [
            e for e in front_loop
            if e not in used_edges and cmds.objExists(e) and _is_open_border_edge(e)
        ]

        back_edges = [
            e for e in best_back
            if e not in used_edges and cmds.objExists(e) and _is_open_border_edge(e)
        ]

        dbg("Bridge candidate -> front: {} | back: {}".format(
            len(front_edges), len(back_edges))
        )

        if not front_edges or not back_edges:
            dbg("Bridge skip : edges invalides ou déjà utilisées.")
            remaining_back.remove(best_back)
            continue

        if len(front_edges) != len(back_edges):
            dbg("Bridge skip : edge count différent.")
            remaining_back.remove(best_back)
            continue

        try:
            cmds.select(front_edges + back_edges, r=True)

            bridge = cmds.polyBridgeEdge(
                front_edges + back_edges,
                divisions=0,
                twist=0,
                taper=1,
                curveType=0
            )[0]

            try:
                cmds.setAttr(bridge + ".sourceDirection", 0)
                dbg("Bridge direction forcée : 0")
            except Exception as e:
                dbg("Impossible de setter sourceDirection : {}".format(e))

            created_bridge_nodes.append(bridge)

            for e in front_edges + back_edges:
                used_edges.add(e)

            dbg("Bridge créé : {}".format(bridge))

        except Exception as e:
            dbg("Bridge échoué : {}".format(e))

        remaining_back.remove(best_back)

    state["bridge_nodes"] = created_bridge_nodes
    cmds.select(clear=True)

    if not created_bridge_nodes:
        cmds.warning("Aucun bridge valide n'a pu être créé.")
    else:
        dbg("Total bridge nodes : {}".format(len(created_bridge_nodes)))

def panel_bevel(short_name, state):
    source_set = (
        short_name + "_panelBackFaces_set"
        if state["is_negative"]
        else short_name + "_panelFrontFaces_set"
    )

    faces = cmds.ls(cmds.sets(source_set, q=True), fl=True) or []
    faces = [f for f in faces if ".f[" in f and cmds.objExists(f)]

    if not faces:
        cmds.warning("Aucune face trouvée pour le bevel.")
        return

    cmds.select(faces, r=True)

    try:
        mel.eval("ShrinkPolygonSelectionRegion;")
    except Exception:
        cmds.warning("ShrinkPolygonSelectionRegion a échoué.")
        return

    bevel_faces = cmds.ls(sl=True, fl=True) or []
    bevel_faces = [f for f in bevel_faces if ".f[" in f and cmds.objExists(f)]

    if not bevel_faces:
        cmds.warning("Aucune face après shrink pour le bevel.")
        return

    bevel_set = short_name + "_panelBevelFaces_set"
    if cmds.objExists(bevel_set):
        cmds.delete(bevel_set)
    cmds.sets(bevel_faces, name=bevel_set)

    cmds.select(bevel_faces, r=True)
    mel.eval("ConvertSelectionToEdgePerimeter;")

    bevel_edges = cmds.ls(sl=True, fl=True) or []
    bevel_edges = [e for e in bevel_edges if ".e[" in e and cmds.objExists(e)]

    dbg("Bevel faces : {}".format(len(bevel_faces)))
    dbg("Bevel edges : {}".format(len(bevel_edges)))

    if not bevel_edges:
        cmds.warning("Aucun edge trouvé pour le bevel.")
        return

    try:
        if state.get("bevel_node") and cmds.objExists(state["bevel_node"]):
            cmds.delete(state["bevel_node"])
            state["bevel_node"] = None

        cmds.select(bevel_edges, r=True)

        bevel_node = cmds.polyBevel3(bevel_edges)[0]
        state["bevel_node"] = bevel_node

        _apply_bevel_state(bevel_node, state)

        dbg("Bevel node créé : {}".format(bevel_node))

    except Exception as e:
        cmds.warning("Bevel failed: {}".format(e))
        return

    cmds.select(clear=True)


def panel_force_initial_bevel_refresh(short_name, state):
    panel_rebuild_bevel(short_name, state)


def panel_rebuild_bevel(short_name, state):
    _store_bevel_state_from_node(state)

    if state.get("bevel_node") and cmds.objExists(state["bevel_node"]):
        try:
            cmds.delete(state["bevel_node"])
        except Exception:
            pass
        state["bevel_node"] = None

    panel_bevel(short_name, state)

def panel_delete_opposite_faces(state):
    if not state["started"]:
        cmds.warning("Lance d'abord Start.")
        return

    faces = _get_opposite_faces_for_current_bevel_side(state)

    shrink_count = max(0, int(state.get("bevel_segments", 1)))

    for _ in range(shrink_count):

            mel.eval("ShrinkPolygonSelectionRegion;")
        except Exception:
            break

    final_faces = [f for f in final_faces if ".f[" in f and cmds.objExists(f)]

    if not final_faces:
        cmds.warning("Aucune face à supprimer après shrink.")
        return

    cmds.delete(final_faces)
    print("Faces opposées supprimées.")


def panel_grow_delete_opposite_faces(state):

    faces = _get_opposite_faces_for_current_bevel_side(state)

def _get_opposite_faces_for_current_bevel_side(state):

        return []

    if state["is_negative"]:
        set_name = short_name + "_panelFrontFaces_set"
    else:
        set_name = short_name + "_panelBackFaces_set"

    faces = cmds.ls(cmds.sets(set_name, q=True), fl=True) or []
    return [f for f in faces if ".f[" in f and cmds.objExists(f)]

def panel_validate(state):
    if not state["started"]:
        cmds.warning("Rien à valider.")
        return

    obj = state.get("obj")
    short_name = state.get("short_name")
    backup = state.get("backup")

    if not obj or not cmds.objExists(obj):
        cmds.warning("Objet final introuvable.")
        return

    temp_nodes = [
        short_name + "_panelEdges_set",
        short_name + "_panelFrontFaces_set",
        short_name + "_panelBackFaces_set",
        short_name + "_panelSideFaces_set",
        short_name + "_panelSeams_set",
        short_name + "_panelBevelFaces_set"
    ]

    for node in temp_nodes:
        if cmds.objExists(node):
            try:
                cmds.delete(node)
            except Exception:
                pass

    if backup and cmds.objExists(backup):
        try:
            cmds.delete(backup)
        except Exception:
            pass

    try:
        cmds.delete(obj, ch=True)
    except Exception:
        pass

    state["backup"] = None
    state["bridge_nodes"] = []
    state["bevel_node"] = None

    cmds.select(obj, r=True)
    print("Panel tool validé et clean.")
    show_info("Validate", "Panel tool validé et historique nettoyé.", [obj])


def panel_revert(state):
    backup = state.get("backup")
    obj = state.get("obj")
    short_name = state.get("short_name")

    if not backup or not cmds.objExists(backup):
        cmds.warning("Backup introuvable.")
        return

    if obj and cmds.objExists(obj):
        try:
            cmds.delete(obj)
        except Exception:
            cmds.warning("Impossible de supprimer l'objet courant.")
            return

    restored = cmds.duplicate(backup, rr=True)[0]
    restored = cmds.rename(restored, short_name)

    try:
        cmds.showHidden(restored)
    except Exception:
        pass

    try:
        cmds.delete(backup)
    except Exception:
        pass

    temp_nodes = [
        short_name + "_panelEdges_set",
        short_name + "_panelFrontFaces_set",
        short_name + "_panelBackFaces_set",
        short_name + "_panelSideFaces_set",
        short_name + "_panelSeams_set",
        short_name + "_panelBevelFaces_set"
    ]

    for node in temp_nodes:
        if cmds.objExists(node):
            try:
                cmds.delete(node)
            except Exception:
                pass

    state["started"] = False
    state["obj"] = restored
    state["short_name"] = restored.split("|")[-1]
    state["backup"] = None
    state["extrude_node"] = None
    state["bridge_nodes"] = []
    state["bevel_node"] = None
    state["safe_selection"] = [restored]

    panel_enable_controls(state, False)
    cmds.select(restored, r=True)

    print("Revert terminé.")
    show_info("Revert", "Mesh restauré depuis le backup.", [restored])
